1. Install and load the necessary packages:

install.packages("rgbif")   # for accessing GBIF data
install.packages("leaflet") # for mapping data
library(rgbif)
library(leaflet)

2. Search GBIF for sage grouse occurrences in Wyoming:

wyoming <- name_backbone("Centrocercus urophasianus",
                          country = "US",
                          hasCoordinate = TRUE,
                          stateProvince = "Wyoming")

3. Extract latitude and longitude coordinates:

wyoming_coords <- wyoming$data[, c("decimalLatitude", "decimalLongitude")]

4. Create an interactive heatmap using Leaflet:

leaflet(wyoming_coords) %>% 
  addTiles() %>%
  addHeatmap(lng = ~decimalLongitude,
             lat = ~decimalLatitude,
             blur = 20,
             max = 0.1)
